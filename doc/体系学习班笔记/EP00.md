# 关于程序输入参数的示例：

## 非文件操作

```csharp
//第一种，一行内含有多个参数，多次调用
string? line;
while (!string.IsNullOrEmpty(line = Console.ReadLine()))
{
    string[] tokens = line.Split(' ');
    Console.WriteLine(int.Parse(tokens[0]) + int.Parse(tokens[1]));
}
//第二种，需要先输入一个raw行，col列的一个矩阵
string? temp = Console.ReadLine();
if (string.IsNullOrEmpty(temp))
    throw new ArgumentNullException();
int row = int.Parse(temp.Trim());

temp = Console.ReadLine();
if (string.IsNullOrEmpty(temp))
    throw new ArgumentNullException();

int col = int.Parse(temp.Trim());
int ans = 0;
for (int i = 0; i < row; i++)
{
    temp = Console.ReadLine();
    if (string.IsNullOrEmpty(temp))
        throw new ArgumentNullException();
    string[]? inputs = temp.Trim().Split(' ');
    for (int j = 0; j < col; j++)
    {
        ans += int.Parse(inputs[j]);
    }
}
Console.WriteLine(ans);
```

## 从文件中读取

```csharp
//读取文件
using var reader = new StreamReader(new FileStream(@"D:\input.txt",  FileMode.Open));
//写入文件
using var writer = new StreamWriter(new FileStream(@"D:\output.txt",  FileMode.OpenOrCreate));

while (!reader.EndOfStream)
{
    _n = int.Parse(reader.ReadLine()!);
    _m = int.Parse(reader.ReadLine()!);
    var paths = Array.ConvertAll(reader.ReadLine()!.Split(), int.Parse);
    for (var i = 0; i < _m; i++) Path[i] = paths[i] - 1;

    for (var i = 0; i < _n; i++)
    {
        var distances = Array.ConvertAll(reader.ReadLine()!.Split(), int.Parse);
        for (var j = 0; j < _n; j++) Distance[i, j] = distances[j];
    }

    _ans = 0;
    for (var i = 1; i < _m; i++) _ans += Distance[Path[i - 1], Path[i]];

    writer.WriteLine(_ans);
}

writer.Flush();
```
