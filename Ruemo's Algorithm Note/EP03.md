**单向链表反转**
~~~c#
public static SingleLinkedNode Reverse()
{
        //将链表分为两个部分:已反转链表和未反转链表
        //next指向未反转链表的头节点
        //pre指向已反转链表的头节点
        var pre = head;
        while (head != null)
        {
            //缩短next指向的未反转链表
            var next = head.Next;
            //让原本的下一个节点指向上一个节点
            head.Next = pre;
            //让pre指向已反转链表的头节点
            pre = head;
            //向右扩展已反转区
            head = next;
        }

        return pre; //返回最终已反转链表的头节点
}
~~~

**单向链表删除节点**
~~~C#
public static SingleLinkedNode Remove(SingleLinkedNode? head, int num){
    // head来到第一个不需要删的位置
    while (head != null)
    {
        if (head.Value != num)
        {
            break;
        }

        head = head.Next;
    }

    // 1 ) head == null
    // 2 ) head != null
    SingleLinkedNode? pre = head;
    SingleLinkedNode? cur = head;
    while (cur != null)
    {
        if (pre != null && cur.Value == num)
        {
            pre.Next = cur.Next;
        }
        else
        {
            pre = cur;
        }

        cur = cur.Next;
    }

    return head;
}
~~~

~~~C#
public void Remove(int num)  
{  
    DoubleLinkedNode<T>? current = Head;  
    while (current != null)  
    {  
        if (current.Data.Equals(num))  
        {  
            // 如果当前节点是头节点  
            if (current == Head)  
            {  
                Head = current.Next;  
                if (Head != null) // 确保下一个节点存在  
                {  
                    Head.Previous = null; // 更新新头节点的Previous  
                }  
            }  
            else  
            {  
                // 否则，更新前一个节点的Next  
                current.Previous?.Next = current.Next;  
            }  

            // 如果当前节点有下一个节点，则更新下一个节点的Previous  
            current.Next?.Previous = current.Previous;  

            // 断开当前节点的引用，以便垃圾收集器可以回收它  
            current = null;  
        }  
        else  
        {  
            // 继续遍历链表  
            current = current.Next;  
        }  
    }  
}
~~~

**栈**
* 数组实现
~~~C#
using System;  
  
public class ArrayStack  
{  
    private int[] elements; // 存储栈元素的数组  
    private int count; // 当前栈中元素的数量  
    private const int initialCapacity = 16; // 栈的初始容量  
  
    // 构造函数  
    public ArrayStack()  
    {  
        elements = new int[initialCapacity];  
        count = 0;  
    }  
  
    // 入栈操作  
    public void Push(int item)  
    {  
        if (count == elements.Length)  
        {  
            // 如果栈已满，则扩容  
            Array.Resize(ref elements, elements.Length * 2);  
        }  
  
        elements[count] = item;  
        count++;  
    }  
  
    // 出栈操作  
    public int Pop()  
    {  
        if (count == 0)  
        {  
            throw new InvalidOperationException("Stack is empty.");  
        }  
  
        count--;  
        int item = elements[count];  
        elements[count] = 0; // 清除元素，可选操作  
  
        // 如果栈中元素数量较少，可以考虑缩容  
        if (count > 0 && count == elements.Length / 4)  
        {  
            Array.Resize(ref elements, elements.Length / 2);  
        }  
  
        return item;  
    }  
  
    // 查看栈顶元素  
    public int Peek()  
    {  
        if (count == 0)  
        {  
            throw new InvalidOperationException("Stack is empty.");  
        }  
  
        return elements[count - 1];  
    }  
  
    // 检查栈是否为空  
    public bool IsEmpty()  
    {  
        return count == 0;  
    }  
  
    // 获取栈的大小  
    public int Count  
    {  
        get { return count; }  
    }  
}  
  
class Program  
{  
    static void Main()  
    {  
        ArrayStack stack = new ArrayStack();  
  
        // 入栈操作  
        stack.Push(1);  
        stack.Push(2);  
        stack.Push(3);  
  
        // 查看栈顶元素  
        Console.WriteLine(stack.Peek()); // 输出: 3  
  
        // 出栈操作  
        Console.WriteLine(stack.Pop()); // 输出: 3  
        Console.WriteLine(stack.Pop()); // 输出: 2  
  
        // 检查栈是否为空  
        Console.WriteLine(stack.IsEmpty()); // 输出: False  
  
        // 获取栈的大小  
        Console.WriteLine(stack.Count); // 输出: 1  
    }  
}
~~~

* 双向链表实现
~~~C#
using System;  
  
public class Node  
{  
    public int Data { get; set; }  
    public Node Next { get; set; }  
    public Node Previous { get; set; }  
}  
  
public class DoublyLinkedListStack  
{  
    private Node top; // 栈顶  
    private int count; // 栈中元素数量  
  
    public DoublyLinkedListStack()  
    {  
        top = null;  
        count = 0;  
    }  
  
    // 入栈操作  
    public void Push(int data)  
    {  
        Node newNode = new Node { Data = data };  
  
        if (top == null)  
        {  
            top = newNode;  
        }  
        else  
        {  
            newNode.Previous = top;  
            top.Next = newNode;  
            top = newNode;  
        }  
  
        count++;  
    }  
  
    // 出栈操作  
    public int Pop()  
    {  
        if (top == null)  
            throw new InvalidOperationException("Stack is empty.");  
  
        int data = top.Data;  
  
        if (top.Next == null)  
        {  
            top = null;  
        }  
        else  
        {  
            top = top.Next;  
            top.Previous = null;  
        }  
  
        count--;  
        return data;  
    }  
  
    // 查看栈顶元素  
    public int Peek()  
    {  
        if (top == null)  
            throw new InvalidOperationException("Stack is empty.");  
  
        return top.Data;  
    }  
  
    // 检查栈是否为空  
    public bool IsEmpty()  
    {  
        return top == null;  
    }  
  
    // 获取栈的大小  
    public int Count()  
    {  
        return count;  
    }  
}  
  
class Program  
{  
    static void Main()  
    {  
        DoublyLinkedListStack stack = new DoublyLinkedListStack();  
  
        // 入栈操作  
        stack.Push(1);  
        stack.Push(2);  
        stack.Push(3);  
  
        // 查看栈顶元素  
        Console.WriteLine(stack.Peek()); // 输出: 3  
  
        // 出栈操作  
        Console.WriteLine(stack.Pop()); // 输出: 3  
        Console.WriteLine(stack.Pop()); // 输出: 2  
  
        // 检查栈是否为空  
        Console.WriteLine(stack.IsEmpty()); // 输出: False  
  
        // 获取栈的大小  
        Console.WriteLine(stack.Count()); // 输出: 1  
    }  
}
~~~

**队列**
* 数组实现
~~~C#
public class ArrayQueue<T>  
{  
    private T[] items;  
    private int head; // 指向队列头部  
    private int tail; // 指向队列尾部的下一个位置  
    private int count; // 队列中的元素数量  
  
    public ArrayQueue(int capacity)  
    {  
        items = new T[capacity];  
    }  
  
    // 入队操作  
    public void Enqueue(T item)  
    {  
        if (count == items.Length)  
        {  
            // 如果队列已满，则扩大数组容量  
            Resize();  
        }  
        items[tail] = item;  
        tail = (tail + 1) % items.Length; // 使用循环队列的方式处理尾部指针  
        count++;  
    }  
  
    // 出队操作  
    public T Dequeue()  
    {  
        if (count == 0)  
        {  
            throw new InvalidOperationException("Queue is empty.");  
        }  
        T item = items[head];  
        items[head] = default(T); // 可选：将出队元素设置为默认值  
        head = (head + 1) % items.Length; // 使用循环队列的方式处理头部指针  
        count--;  
        return item;  
    }  
  
    // 查看队列头部元素，不移除  
    public T Peek()  
    {  
        if (count == 0)  
        {  
            throw new InvalidOperationException("Queue is empty.");  
        }  
        return items[head];  
    }  
  
    // 检查队列是否为空  
    public bool IsEmpty()  
    {  
        return count == 0;  
    }  
  
    // 扩大数组容量  
    private void Resize()  
    {  
        int newSize = items.Length * 2; // 可以根据需要调整扩大策略  
        T[] newItems = new T[newSize];  
        for (int i = 0; i < count; i++)  
        {  
            int index = (head + i) % items.Length; // 处理循环队列的索引计算  
            newItems[i] = items[index]; // 将旧数组中的元素复制到新数组  
        }  
        head = 0; // 重置头部指针  
        tail = count; // 重置尾部指针，指向新位置的下一个位置（因为可能有元素已经出队）  
        items = newItems; // 替换底层数组  
    }  
}
~~~

* 双向链表实现
~~~C#
using System;  
  
public class Node  
{  
    public int Data { get; set; }  
    public Node Next { get; set; }  
    public Node Previous { get; set; }  
}  
  
public class DoublyLinkedListQueue  
{  
    private Node front; // 队首  
    private Node rear;  // 队尾  
  
    public DoublyLinkedListQueue()  
    {  
        front = null;  
        rear = null;  
    }  
  
    // 入队操作  
    public void Enqueue(int data)  
    {  
        Node newNode = new Node { Data = data };  
  
        if (rear == null)  
        {  
            front = newNode;  
            rear = newNode;  
        }  
        else  
        {  
            rear.Next = newNode;  
            newNode.Previous = rear;  
            rear = newNode;  
        }  
    }  
  
    // 出队操作  
    public int Dequeue()  
    {  
        if (front == null)  
            throw new InvalidOperationException("Queue is empty.");  
  
        int data = front.Data;  
  
        if (front == rear)  
        {  
            front = null;  
            rear = null;  
        }  
        else  
        {  
            front = front.Next;  
            front.Previous = null;  
        }  
  
        return data;  
    }  
  
    // 检查队列是否为空  
    public bool IsEmpty()  
    {  
        return front == null;  
    }  
  
    // 获取队列的大小  
    public int Size()  
    {  
        int count = 0;  
        Node current = front;  
        while (current != null)  
        {  
            count++;  
            current = current.Next;  
        }  
        return count;  
    }  
}  
  
class Program  
{  
    static void Main()  
    {  
        DoublyLinkedListQueue queue = new DoublyLinkedListQueue();  
  
        // 入队操作  
        queue.Enqueue(1);  
        queue.Enqueue(2);  
        queue.Enqueue(3);  
  
        // 出队操作  
        Console.WriteLine(queue.Dequeue()); // 输出: 1  
        Console.WriteLine(queue.Dequeue()); // 输出: 2  
  
        // 检查队列大小  
        Console.WriteLine(queue.Size()); // 输出: 1  
    }  
}
~~~

题目二：如何用栈实现队列结构

~~~C#
//重点:当需要从队列中出队时需要将push栈向pop栈倒入数据
private class TwoStacksQueue
{
    private readonly Stack<Int32> _stackPush;
    private readonly Stack<Int32> _stackPop;

    public TwoStacksQueue()
    {
        _stackPush = new Stack<Int32>();
        _stackPop = new Stack<Int32>();
    }

    // push栈向pop栈倒入数据
    //注意：
    //不能只倒入一部分！！！
    //需要pop栈清空才能重新倒入数据
    private void PushToPop()
    {
        if (_stackPop.Count == 0)
        {
            while (_stackPush.Count != 0)
            {
                _stackPop.Push(_stackPush.Pop());
            }
        }
    }

    public void Add(int pushInt)
    {
        _stackPush.Push(pushInt);
        PushToPop();
    }

    public int Poll()
    {
        if (_stackPop.Count == 0 && _stackPush.Count == 0)
        {
            throw new Exception("Queue is empty!");
        }

        PushToPop();
        return _stackPop.Pop();
    }

    public int Peek()
    {
        if (_stackPop.Count == 0 && _stackPush.Count == 0)
        {
            throw new Exception("Queue is empty!");
        }

        PushToPop();
        return _stackPop.Peek();
    }
}
~~~

题目三：如何用队列实现栈结构

~~~C#
//重点:需要出栈时需要将除队头元素的其他元素移动到另一个空队列中，将原来队列剩下的元素返回并清空
private class TwoQueueStack<T>
{
    private Queue<T> _queue;
    private Queue<T> _help;

    public TwoQueueStack()
    {
        _queue = new Queue<T>();
        _help = new Queue<T>();
    }

    public void Push(T value)
    {
        _queue.Enqueue(value);
    }

    public T Dequeue()
    {
        while (_queue.Count > 1)
        {
            _help.Enqueue(_queue.Dequeue());
        }

        T ans = _queue.Dequeue();
        (_queue, _help) = (_help, _queue);
        return ans;
    }

    public T Peek()
    {
        while (_queue.Count > 1)
        {
            _help.Enqueue(_queue.Dequeue());
        }

        T ans = _queue.Dequeue();
        _help.Enqueue(ans);
        (_queue, _help) = (_help, _queue);
        return ans;
    }

    public bool IsEmpty()
    {
        return _queue.Count == 0;
    }
}
~~~

**重写比较器**

在C#中，比较器（Comparer）通常用于定义自定义的排序规则，它们经常用在集合类如`List<T>`, `Array`, `Dictionary<TKey, TValue>`等中。C#提供了几种不同的方式来定义比较器。以下是几种常见的比较器写法：

1. **使用内置的`IComparer<T>`接口**:
   实现`IComparer<T>`接口是最基本的方式，你需要提供`Compare`方法，该方法接受两个`T`类型的参数并返回一个整数，表示它们的比较结果。

```csharp
public class PersonComparer : IComparer<Person>  
{  
    public int Compare(Person x, Person y)  
    {  
        return x.Age.CompareTo(y.Age);  
    }  
}
```

2. **使用`Comparison<T>`委托**:
   `Comparison<T>`是一个内置委托，它也可以用于提供自定义的比较逻辑。它通常用在LINQ查询和集合类的排序方法中。

```csharp
Comparison<Person> personComparison = (x, y) => x.Age.CompareTo(y.Age);  
List<Person> people = new List<Person>();  
// ... 填充people列表 ...  
people.Sort(personComparison);
```

3. **使用Lambda表达式**:
   在C#中，你可以使用Lambda表达式作为匿名方法来提供比较逻辑，这在需要对集合进行一次性排序时非常方便。

```csharp
List<Person> people = new List<Person>();  
// ... 填充people列表 ...  
people.Sort((x, y) => x.Age.CompareTo(y.Age));
```

4. **使用`IComparer`接口（非泛型）**:
   对于非泛型集合，你可能需要实现`IComparer`接口，它要求实现`Compare`方法，该方法接受两个`Object`类型的参数。

```csharp
public class PersonComparer : IComparer  
{  
    public int Compare(object x, object y)  
    {  
        if (x is Person personX && y is Person personY)  
        {  
            return personX.Age.CompareTo(personY.Age);  
        }  
        throw new ArgumentException("Objects must be of type Person");  
    }  
}
```

5. **使用`IComparable<T>`接口**:
   如果你想让你的类型能够自己定义比较逻辑，你可以实现`IComparable<T>`接口。这样，你的类型的实例就可以用在任何需要`IComparer<T>`的地方。

```csharp
public class Person : IComparable<Person>  
{  
    public int Age { get; set; }  
  
    public int CompareTo(Person other)  
    {  
        return this.Age.CompareTo(other.Age);  
    }  
}
```

6. **使用`IEqualityComparer<T>`接口**:
   如果你需要定义相等性比较逻辑，你可以实现`IEqualityComparer<T>`接口，它包含`Equals`和`GetHashCode`两个方法。

```csharp
public class PersonEqualityComparer : IEqualityComparer<Person>  
{  
    public bool Equals(Person x, Person y)  
    {  
        if (ReferenceEquals(x, y)) return true;  
        if (ReferenceEquals(x, null) || ReferenceEquals(y, null)) return false;  
        return x.Age == y.Age;  
    }  
  
    public int GetHashCode(Person obj)  
    {  
        return obj.Age.GetHashCode();  
    }  
}
```

每种方法都有其适用的场景，你可以根据具体需求选择最合适的方式来实现比较器。在创建比较器时，务必注意正确处理边界情况，如`null`值、类型不匹配等，以避免运行时错误。
